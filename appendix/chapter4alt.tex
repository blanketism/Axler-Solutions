\subsection{Polynomials}

\begin{apex}[4a9]
    Prove that every polynomial of odd degree with real coefficients has a real zero.
\end{apex}

\begin{proof}[blanketism]
    Let $p(x)$ be a polynomial of odd degree with real coefficients.
	As $x \to \infty$, $p(x) \to \infty$ or $p(x) \to -\infty$ depending on the leading coefficient.
	Similarly, as $x \to -\infty$, $p(x) \to -\infty$ or $p(x) \to \infty$.
	By the intermediate value theorem, since $p(x)$ is continuous and takes on both positive and negative values, there must exist some $x \in \r$ such that $p(x) = 0$.
	Therefore, $p$ has a real zero.
\end{proof}

\begin{apex}[4a11]
    Suppose $p \in \pp(\c)$.
	Define $q: \c \to \c$ by
    \[q(z) = p(z) \bar{p(\bar{z})}.\]
    Prove that $q$ is a polynomial with real coefficients.
\end{apex}

\begin{proof}[blanketism]
    If $p = 0$, then $q = 0$ is a polynomial with real coefficients.
	Otherwise, let $n = \deg p$.
	Then we may write
    \[p(z) = \alpha(z - \lambda_1)(z - \lambda_2) \cdots (z - \lambda_n),\]
    where $\alpha \in \c$ and $\lambda_1, \lambda_2, \ldots, \lambda_n \in \c$ are the roots of $p$.
	Then
    \[q(z) = p(z) \bar{p(\bar{z})} = \alpha(z - \lambda_1)(z - \lambda_2) \cdots (z - \lambda_n) \cdot \bar{\alpha} (z - \bar{\lambda_1}) (z - \bar{\lambda_2}) \cdots (z - \bar{\lambda_n}).\]
    Note that the product contains the term $\alpha\bar\alpha = |\alpha|^2 \in \r$, as well as the pairs
    \[(z - \lambda_i)(z - \bar{\lambda_i}) = z^2 - 2\Re(\lambda_i) z + |\lambda_i|^2.\]
    Each such pair contributes a quadratic polynomial with real coefficients.
	Since the product of polynomials with real coefficients also has real coefficients, it follows that $q(z)$ is a polynomial with real coefficients.
\end{proof}
